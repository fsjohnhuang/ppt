Imperative vs. Declarative



Imperative way

<div id="signup">
  <label>Name:</label><input name="name" placeholder="Enter Name.">
  <input type="submit" value="Submit" onclick="submit("#signup")">
</div>
function submit(containerId){
  var kvs = {}
  $(containerId).find('input').forEach(el => kvs[el.name] = el.value)
  $.post({
    url: './users',
    data: kvs
  })
}

Declarative way

<x-form action="./users" method="POST">
    <label>Name:</label><input name="name" placeholder="Enter Name.">
    <input type="submit" value="Submit">    
</x-form>
